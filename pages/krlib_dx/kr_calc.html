使用例

ランダムな方向に移動する。 


              //ランダムな角度(0～360度)
              double ang = Calc::RandNum(0, 360, true);
              //角度 → ベクトルに変換.
              DBL_XY dir = Calc::AngToVector(ang);
              //移動.
              pos += dir * 5.0;

座標計算系。

              //距離を取得.
              double distance = Calc::Dist(posA, posB);
