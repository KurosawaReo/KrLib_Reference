使用例

main関数で InitDx LoopDx を実行します。
ManagerBase を継承した管理クラスの Init Reset Update Draw 関数が自動で呼ばれるようになります。 


              int WINAPI WinMain(
                _In_     HINSTANCE hinstance,
                _In_opt_ HINSTANCE hPrevinstance,
                _In_     LPSTR     lpCmdLine,
                _In_     int       nCmdShow
              ){
                try {
                  //初期化処理.
                  App::InitDx(WINDOW_WID, WINDOW_HEI, IS_WINDOW_MODE, FPS, false);
                }
                catch (const ErrorMsg& err) {
                  //エラーメッセージ.
                  Debug::Log(_T("InitDx"), err.GetResult());
                }

                try {
                  //ループ処理.
                  App::LoopDx();
                }
                catch (const ErrorMsg& err) {
                  //エラーメッセージ.
                  Debug::Log(_T("LoopDx"), err.GetResult());
                }

                return 0;
              }
